        body ->  padding-left: 158px;
                 padding-right: 166px;       
        left-fixed -> width: 150px;
                      float: left;
                      margin-left: calc(-100% - 4px);
                      position: relative;
                      left: -162;
        center-auto -> width: 100%;
                       float: left;
        right-fixed -> width: 150px;
                       float: left; float: left;
                       margin-left: -166px;
                       position: relative;
                       right: -166px;
